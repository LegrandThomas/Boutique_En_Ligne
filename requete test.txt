Select nom,prenom,adresse,code_postal,ville,pays from client 
inner join adresse on (adresse.id_adresse=client.id_adresse); 

SELECT * FROM `variant_produit`
inner join matiere as m on (m.id_matiere=variant_produit.id_matiere)
where valeur='nylon'


select * from commande 
join client on (client.id_utilisateur=commande.id_utilisateur) 
join details_commande on (details_commande.id_commande=commande.id_commande) 
GROUP by client.id_utilisateur; 

SELECT nom,description,valeur,prix FROM produit
inner join decliner on(decliner.id_produit=produit.id_produit)
inner join couleur on (couleur.id_couleur=decliner.id_couleur)               
where decliner.id_produit=1;

SELECT nom,description,nom_entrepot,prix,quantite_actuelle,quantite_minimun from produit
inner join stock on (produit.id_stock=stock.id_stock);


select * from produit
inner join illustrer on ( produit.id_produit=illustrer.id_produit)
join image on (image.id_image=illustrer.id_image)
where produit.nom='pull';

inner join illustrer on ( produit.id_produit=illustrer.id_produit)

SELECT * from moyenpaiement

SELECT * from avis
join client on (client.id_utilisateur=avis.id_utilisateur)
join produit on (produit.id_produit=avis.id_produit)
where client.nom='chirac';
